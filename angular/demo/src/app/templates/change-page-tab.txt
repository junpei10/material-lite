<nav class='docs-page-tab'>
  <button mlButton [theme]='pageName === "overview" ? "primary" : ""' routerLink='overview'>Overview</button>
  <button mlButton [theme]='pageName === "reference" ? "primary" : ""' routerLink='reference'>Reference</button>
  <button mlButton [theme]='pageName === "example" ? "primary" : ""' routerLink='example'>Example</button>
</nav>

onRouteChange(event): void {
  this.pageName = event instanceof OverviewComponent
    ? 'overview'
    : event instanceof ReferenceComponent
      ? 'reference'
      : 'example';
}
